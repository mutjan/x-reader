    3. 添加交互函数（在现有事件函数区域）：
    ```javascript
    function toggleEventExpand(eventId) {
        if (expandedEvents.has(eventId)) {
            expandedEvents.delete(eventId);
        } else {
            expandedEvents.add(eventId);
        }
        renderEvents();
    }

    function toggleAllEvents() {
        const filteredEvents = getFilteredAndSortedEvents();
        const allExpanded = filteredEvents.every(e => expandedEvents.has(e.id));

        if (allExpanded) {
            filteredEvents.forEach(e => expandedEvents.delete(e.id));
        } else {
            filteredEvents.forEach(e => expandedEvents.add(e.id));
        }
        renderEvents();
    }

    function toggleEventSelection(eventId) {
        // 事件级别选择，后续可用于批量操作
        console.log('Event selected:', eventId);
    }

    function toggleSelectAllEvents(checkbox) {
        // 全选/取消全选事件
        console.log('Select all events:', checkbox.checked);
    }
    ```
