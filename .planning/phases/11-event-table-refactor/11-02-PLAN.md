    ```javascript
    function renderEventsTable(events, container) {
        const ratingClass = (rating) => `rating-${rating || 'C'}`;

        let html = `
            <table class="event-table">
    ```

    4. 实现交互函数：
    ```javascript
    function toggleEventExpand(eventId) {
        if (expandedEvents.has(eventId)) {
            expandedEvents.delete(eventId);
        } else {
            expandedEvents.add(eventId);
        }
        renderEvents(currentEventsData || []);
    }

    function toggleEventSelection(eventId) {
        if (selectedEvents.has(eventId)) {
            selectedEvents.delete(eventId);
        } else {
            selectedEvents.add(eventId);
        }
        renderEvents(currentEventsData || []);
    }

    function toggleSelectAllEvents(checkbox) {
        const events = currentEventsData || [];
        if (checkbox.checked) {
            events.forEach(e => selectedEvents.add(e.event_id));
        } else {
            selectedEvents.clear();
        }
        renderEvents(events);
    }
    ```
